1.安装

执行下方dockerfile

version: "3"

services:
  unblockmuisic:
    image: pan93412/unblock-netease-music-enhanced:latest
    restart: always
    command: -s -o kuwo kugou bilibili -e https://music.163.com
    ports:
        - 8080:8080
        - 8081:8081
    environment:
        - JSON_LOG=true
        - LOG_LEVEL=debug
        - ENABLE_FLAC=true
        - ENABLE_LOCAL_VIP=true
